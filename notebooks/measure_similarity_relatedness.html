def plot_model_comp(similarity_func):
    df_func = df_embeddings[df_embeddings["similarity_func"] == similarity_func]
    df_func.loc[:, "model_name"] = df_func["model_name"].map(model_names_mapper)

    fig, axes = plt.subplots(1, 2, figsize=(15, 5))
    df = df_func[["model_name", "metric", "similarity_func", "similarity_cor", "relatedness_cor"]]
    df = df.reset_index()
    df = df.sort_values(by="model_name")
    sns.barplot(ax=axes[0], x="model_name", y="similarity_cor", hue="metric", data=df)
    axes[0].set_title("Similarity correlation per model using {}".format(similarity_func))
    axes[0].set_xticklabels(
        axes[0].get_xticklabels(), 
        rotation=45, 
        horizontalalignment='right',
        fontweight='light',
        fontsize='x-large'
    )
    
    df = df_func[["model_name", "metric", "similarity_func", "similarity_cor", "relatedness_cor"]].groupby(["model_name", "metric", "similarity_func"]).max()
    df = df.reset_index()
    df = df.sort_values(by="model_name")
    sns.barplot(ax=axes[1], x="model_name", y="relatedness_cor", hue="metric", data=df)
    axes[1].set_title("Relatedness correlation per model using {}".format(similarity_func))
    axes[1].set_xticklabels(
        axes[1].get_xticklabels(), 
        rotation=45, 
        horizontalalignment='right',
        fontweight='light',
        fontsize='x-large'
    )

def compare_wordnets(df_wordnets, similarity_func):
    fig, axes = plt.subplots(1, 2, figsize=(15, 5))
    df_wordnets = df_wordnets[df_wordnets["similarity_func"]==similarity_func]
    df = df_wordnets[["model_name", "metric", "similarity_func", "similarity_cor", "relatedness_cor"]].groupby(["metric", "model_name"]).max()
    df = df.reset_index()
    df = df.sort_values(by="model_name")
    sns.barplot(ax=axes[0], x="model_name", y="similarity_cor", hue="metric", data=df)
    axes[0].set_title("Similarity correlation per model using {}".format(similarity_func))
    axes[0].set_xticklabels(
        axes[0].get_xticklabels(), 
        rotation=45, 
        horizontalalignment='right',
        fontweight='light',
        fontsize='x-large'
    )
    
    df = df_wordnets[["model_name", "metric", "similarity_func", "similarity_cor", "relatedness_cor"]].groupby(["metric", "model_name"]).max()
    df = df.reset_index()
    df = df.sort_values(by="model_name")
    sns.barplot(ax=axes[1], x="model_name", y="relatedness_cor", hue="metric", data=df)
    axes[1].set_title("Relatedness correlation per model using {}".format(similarity_func))
    axes[1].set_xticklabels(
        axes[1].get_xticklabels(), 
        rotation=45, 
        horizontalalignment='right',
        fontweight='light',
        fontsize='x-large'
    )
